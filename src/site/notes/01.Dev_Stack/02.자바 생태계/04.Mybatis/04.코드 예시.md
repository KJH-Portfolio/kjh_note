### 1\. Controller (Servlet)

**역할:** 요청(URL)을 받아 분석하고, 적절한 Service 로직을 호출한 뒤, 그 결과를 View(JSP)나 다른 URL로 연결해주는 **교통정리 역할**을 합니다.

```java
// [Annotation] @WebServlet("/detail.bo"): 브라우저 주소창에 /detail.bo 요청이 오면 이 클래스를 실행합니다.
@WebServlet("/detail.bo")
public class BoardDetailController extends HttpServlet {

    // [GET] 주로 데이터를 '조회'해서 보여줄 때 사용합니다.
    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {

        // 1. 파라미터 수신
        // request.getParameter("name")은 무조건 String 타입으로 반환됩니다.
        // 숫자 계산이 필요하다면 Integer.parseInt()로 형변환을 해야 합니다.
        int bno = Integer.parseInt(request.getParameter("bno"));

        // 2. 서비스 호출 (비즈니스 로직)
        // Service 객체를 만들고, 실제 DB에서 게시글 하나를 가져오는 메서드를 시킵니다.
        BoardService service = new BoardService();
        Board b = service.selectBoard(bno);

        // 3. 결과 처리 및 화면 이동
        if(b != null) { 
            // 성공 시: JSP 화면에서 사용할 수 있도록 데이터를 가방(request)에 담습니다.
            // "b"라는 이름표를 붙여서 객체 b를 담음
            request.setAttribute("b", b);

            // [Forward] 포워딩
            // 요청정보(request)와 데이터(b)를 꽉 쥐고 지정한 JSP 파일로 화면만 넘깁니다.
            // 주소창의 URL은 바뀌지 않고 내용물만 바뀝니다.
            request.getRequestDispatcher("/WEB-INF/views/board/boardDetailView.jsp").forward(request, response);
        } else {
            // 실패 시: 에러 메시지를 세션(Session)에 담습니다.
            // 세션은 브라우저가 켜져있는 동안 유지되므로, 페이지가 이동돼도 메시지가 살아있습니다.
            request.getSession().setAttribute("alertMsg", "조회 실패");
            
            // [Redirect] 리다이렉트
            // 브라우저에게 "이 주소로 다시 접속해!"라고 명령합니다.
            // 주소창의 URL이 변경됩니다. (새로운 요청 발생)
            response.sendRedirect(request.getContextPath() + "/list.bo?currentPage=1");
        }
    }

    // [POST] 주로 데이터 '등록/수정/삭제' 등 처리에 사용됩니다.
    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
        
        // POST 방식은 데이터가 몸체(Body)에 숨겨져 오므로 한글이 깨질 수 있습니다.
        // 반드시 인코딩을 UTF-8로 지정해야 합니다.
        request.setCharacterEncoding("UTF-8");

        // JSP의 <form> 태그 안에 있는 input들의 name 속성값으로 데이터를 꺼냅니다.
        String title = request.getParameter("boardTitle");
        String content = request.getParameter("boardContent");
        
        // ... (이후 Service 호출 로직) ...
    }
}
```

### 2\. Service (비즈니스 로직)

**역할:** DB 연결(Connection)을 관리하고, 트랜잭션(Commit/Rollback)을 제어하며, DAO에 일을 시키는 **중간 관리자 역할**입니다.

```java
public class BoardService {

    private BoardDao dao = new BoardDao();

    // 게시글 등록 기능
    public int insertBoard(Board b) {
        
        // 1. Connection 생성 (MyBatis에서는 SqlSession이라고 부름)
        // Template 클래스에서 만들어둔 공장을 통해 연결 객체를 하나 빌려옵니다.
        // (참고: autoCommit은 꺼져있습니다.)
        SqlSession sqlSession = Template.getSqlSession();

        // 2. DAO 호출
        // 연결 객체(sqlSession)와 데이터(b)를 DAO에게 전달해 DB 작업을 시킵니다.
        int result = dao.insertBoard(sqlSession, b);

        // 3. 트랜잭션 처리 (중요)
        // result > 0 이면 DB에 행이 잘 들어갔다는 뜻입니다.
        if(result > 0) {
            sqlSession.commit();   // 확정 (저장)
        } else {
            sqlSession.rollback(); // 취소 (되돌리기)
        }

        // 4. 자원 반납
        // 다 쓴 연결 객체는 반드시 닫아줘야 메모리 누수가 없습니다.
        sqlSession.close();

        return result;
    }
}
```

### 3\. DAO (Data Access Object)

**역할:** 실제 SQL 쿼리가 적혀있는 XML Mapper를 호출하여 **DB에 접근하는 역할**입니다.

```java
public class BoardDao {

    // 게시글 목록 조회
    public ArrayList<Board> selectList(SqlSession sqlSession, PageInfo pi) {
        
        // 페이징 처리용 도구 (RowBounds) 준비
        // offset: 앞에서부터 몇 개 건너뛸지 계산 ((현재페이지-1) * 한페이지당개수)
        // limit: 몇 개 가져올지
        int offset = (pi.getCurrentPage() - 1) * pi.getBoardLimit();
        int limit = pi.getBoardLimit();
        RowBounds rowBounds = new RowBounds(offset, limit);

        // MyBatis에게 명령: "boardMapper"라는 이름표를 가진 파일 안의 "selectList" 쿼리를 실행해줘.
        // 파라미터는 없으니 null, 페이징 정보는 rowBounds에 담아 보냄.
        return (ArrayList)sqlSession.selectList("boardMapper.selectList", null, rowBounds);
    }

    // 게시글 등록
    public int insertBoard(SqlSession sqlSession, Board b) {
        // "boardMapper"의 "insertBoard" 쿼리 실행. 데이터 b를 전달.
        return sqlSession.insert("boardMapper.insertBoard", b);
    }
}
```

### 4\. XML Mapper (`*-mapper.xml`)

```xml
<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN" "https://mybatis.org/dtd/mybatis-3-mapper.dtd">

// namespace: Java(DAO)에서 이 매퍼를 부를 때 쓸 별명입니다.
<mapper namespace="boardMapper">
<!--ㅇㄹㅇㄹㅇ-->
    // [ResultMap 설정]
    // DB의 컬럼명(보통 대문자+언더바)과 자바 객체의 변수명(소문자+낙타표기법)이 다를 때 
    // 서로 짝을 지어주는 지도 역할을 합니다.
    // - column : DB 컬럼명
    // - property : Java 변수명
    <resultMap type="Board" id="boardResultMap">
        <result column="BOARD_NO" property="boardNo"/>
        <result column="BOARD_TITLE" property="boardTitle"/>
        <result column="BOARD_WRITER" property="boardWriter"/>
        <result column="CREATE_DATE" property="createDate"/>
        // 조회할 때 이 규칙대로 데이터를 담아줍니다.
    </resultMap>

    // [SELECT 구문] 
    // - id : 이 쿼리의 고유 이름 (DAO에서 호출함)
    // - resultMap : 조회 결과를 위의 'boardResultMap' 규칙에 따라 객체에 담음
    <select id="selectList" resultMap="boardResultMap">
        SELECT BOARD_NO, BOARD_TITLE, COUNT, CREATE_DATE
        FROM BOARD
        WHERE STATUS = 'Y'
        ORDER BY CREATE_DATE DESC
    </select>

    // [INSERT 구문]
    // - parameterType : 전달받는 데이터의 타입 (Board 객체)
    // - #{변수명} : 전달받은 Board 객체의 get변수명()을 실행해서 값을 채워 넣음
    //   예: #{boardTitle} -> b.getBoardTitle()과 같음
    <insert id="insertBoard" parameterType="Board">
        INSERT INTO BOARD (BOARD_NO, BOARD_TITLE, BOARD_CONTENT, BOARD_WRITER)
        VALUES (
            SEQ_BNO.NEXTVAL,  // 시퀀스로 번호 자동 생성
            #{boardTitle},    
            #{boardContent},  
            #{boardWriter}    
        )
    </insert>

</mapper>
```

### 5\. JSP (Java Server Page) & Frontend

```ejs
<%@ page language="java" contentType="text/html; charset=UTF-8" pageEncoding="UTF-8"%>
<%-- JSTL 코어 라이브러리 선언 --%>
<%@ taglib prefix="c" uri="http://java.sun.com/jsp/jstl/core"%>

<!DOCTYPE html>
<html>
<head>
    <title>게시글 목록</title>
</head>
<body>
    
    <%-- 공통 메뉴바 include --%>
    <%@ include file="/WEB-INF/views/common/menubar.jsp" %>

    <%-- Context Root 경로를 변수(contextRoot)에 저장 --%>
    <c:set var="contextRoot" value="${pageContext.request.contextPath}" />

    <div class="outer">
        <h2>게시판</h2>

        <%-- [조건문] 로그인한 사용자만 글작성 버튼 표시 --%>
        <c:if test="${not empty loginMember}">
            <a href="${contextRoot}/insert.bo">글작성</a>
        </c:if>

        <table>
            <thead>
                <tr>
                    <th>번호</th>
                    <th>제목</th>
                    <th>작성자</th>
                    <th>조회수</th>
                    <th>작성일</th>
                </tr>
            </thead>
            <tbody>
                <%-- [분기문] 게시글 유무에 따라 다른 화면 출력 --%>
                <c:choose>
                    <%-- 1. 리스트가 비어있을 때 --%>
                    <c:when test="${empty list}">
                        <tr><td colspan="5">게시글이 없습니다.</td></tr>
                    </c:when>

                    <%-- 2. 리스트가 있을 때 (반복문) --%>
                    <c:otherwise>
                        <c:forEach items="${list}" var="b">
                            <tr>
                                <td>${b.boardNo}</td>
                                <td>${b.boardTitle}</td>
                                <td>${b.boardWriter}</td>
                                <td>${b.count}</td>
                                <td>${b.createDate}</td>
                            </tr>
                        </c:forEach>
                    </c:otherwise>
                </c:choose>
            </tbody>
        </table>

        <%-- [페이징] 시작 페이지 ~ 끝 페이지 번호 출력 --%>
        <div class="paging-area">
            <c:forEach var="i" begin="${pi.startPage}" end="${pi.endPage}">
                <button onclick="location.href='${contextRoot}/list.bo?currentPage=${i}'">${i}</button>
            </c:forEach>
        </div>
    </div>

    <script>
        <%-- jQuery: 테이블 행 클릭 시 상세보기 이동 --%>
        $(function(){
            $(".list-area tbody tr").click(function(){
                <%-- 클릭된 행의 첫 번째 칸(글번호) 텍스트 가져오기 --%>
                let bno = $(this).children().eq(0).text();
                
                <%-- GET 방식으로 상세 페이지 요청 --%>
                location.href = '${contextRoot}/detail.bo?bno=' + bno;
            });
        });
    </script>
</body>
</html>
```

### 6\. Config (설정 파일)

```xml
<configuration>
    
    <settings>
        // [설정] NULL 처리
        // 데이터 입력 시 값이 없어서 null이 들어갈 때, 
        // Oracle DB 등에서 에러가 나지 않도록 JDBC 타입 NULL로 처리하겠다는 설정 
        <setting name="jdbcTypeForNull" value="NULL"/>
    </settings>
    
    // [별칭 설정] typeAliases
    // 패키지 경로가 긴 클래스(VO)들을 짧은 이름으로 줄여 쓰기 위해 등록합니다.
    // 예: com.kh.mybatis.board.model.vo.Board -> Board 라고만 써도 인식됨
    <typeAliases>
        <typeAlias type="com.kh.mybatis.member.model.vo.Member" alias="Member"/>
        <typeAlias type="com.kh.mybatis.board.model.vo.Board" alias="Board"/>
    </typeAliases>
    
    // [DB 환경 설정] environments
    <environments default="development">
        <environment id="development">
            
            // transactionManager type="JDBC"
            // 트랜잭션(커밋/롤백)을 개발자가 직접 코드로 제어하겠다는 설정입니다. 
            <transactionManager type="JDBC"/>
            
            // dataSource type="POOLED" (Connection Pool 사용)
            // DB 연결 객체를 미리 여러 개 만들어두고 빌려 쓰는 방식입니다.
            // 매번 연결을 새로 만드는 것보다 속도가 훨씬 빠릅니다.
            <dataSource type="POOLED">
                <property name="driver" value="oracle.jdbc.driver.OracleDriver"/>
                <property name="url" value="jdbc:oracle:thin:@localhost:1521:xe"/>
                <property name="username" value="MYBATIS"/>
                <property name="password" value="MYBATIS"/>
            </dataSource>
        </environment>
    </environments>
    
    // [매퍼 등록] mappers
    // SQL 쿼리가 작성된 XML 파일들의 위치를 여기에 등록해야 MyBatis가 인식합니다.
    <mappers>
        <mapper resource="/resources/mappers/member-mapper.xml"/>
        <mapper resource="/resources/mappers/board-mapper.xml"/>
    </mappers>
    
    	<!--setting으로 DB컬럼명을 java객체 필드명으로 자동 변환 (USER_ID -> userId)-->
	<settings>
	    <setting name="logImpl" value="STDOUT_LOGGING"/>
	    <setting name="mapUnderscoreToCamelCase" value="true"/>
	</settings>

</configuration>
```
