```js
import React, { createContext, useContext, useEffect, useState, useRef } from 'react';
import { Client } from '@stomp/stompjs';

const ChatContext = createContext();
export const useChat = () => useContext(ChatContext);

export const ChatProvider = ({ children }) => {
  const [client, setClient] = useState(null);
  const stompClientRef = useRef(null);

  // ✅ [핵심: 발신 로직]
  const sendMessage = (destination, body) => {
    if (stompClientRef.current && stompClientRef.current.connected) {
      stompClientRef.current.publish({
        destination: destination, // 백엔드 @MessageMapping 주소 (예: '/app/chat')
        body: JSON.stringify(body), // 전송할 데이터
      });
    } else {
      console.error("소켓이 연결되지 않았습니다.");
    }
  };

  //클라이언트 설정
  useEffect(() => {
    const stompClient = new Client({
      brokerURL: 'ws://localhost:8080/spring/ws-chat',
      connectHeaders: {
        Authorization: `Bearer ${localStorage.getItem('token')}`,
      },
      reconnectDelay: 5000,
      
      //연결시 설정
      onConnect: () => {
        console.log('STOMP Connected');
        
        // ✅ [핵심: 수신 로직]
        stompClient.subscribe('/topic/some-channel', (message) => {
          console.log('메시지 도착!', JSON.parse(message.body));
        });
      },
    });

    stompClient.activate();
    stompClientRef.current = stompClient;
    setClient(stompClient);

    return () => {
      if (stompClientRef.current) stompClientRef.current.deactivate();
    };
  }, []);

  return (
    // 발신용 sendMessage 함수를 value에 포함시켜서 하위로 전달합니다.
    <ChatContext.Provider value={{ client, sendMessage }}>
      {children}
    </ChatContext.Provider>
  );
};
```
