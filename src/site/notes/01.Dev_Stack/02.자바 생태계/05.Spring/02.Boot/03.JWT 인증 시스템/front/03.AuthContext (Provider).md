# AuthContext 기반 초기 인증 시스템 구성 (AuthProvider
)
> [!summary]
> 
> `Context API`를 통해 전역 인증 상태를 관리하고, 로컬스토리지 기반의 세션 유지 및 기본적인 로그인/로그아웃 기능을 제공하는 초기 버전의 명세임.
---
## 인증 상태 관리 및 초기화
- **상태 변수 구성**
    - **user**: 로그인된 사용자 객체 정보를 저장하는 상태
    - **isLoading**: 인증 정보 복구 프로세스의 진행 여부를 나타내는 플래그
    - **isAuthenticated**: 사용자 인증 완료 여부를 판단하는 불리언 값
- **인증 정보 복구 (useEffect)**
    - 앱 로드 시 로컬스토리지 내 `token` 및 `user` 데이터 존재 여부 확인
    - 데이터 유효성 검증 및 JSON 파싱을 통한 사용자 상태 복구
    - 파싱 오류 발생 시 스토리지 데이터 초기화 및 보안 유지
```javascript
// 인증 관련 컨텍스트 및 프로바이더 컴포넌트 관리
import { createContext, useCallback, useContext, useEffect, useState } from "react";
import authApi from "../api/authApi"; // 인증 관련 API 호출 모듈

// 인증 상태 공유를 위한 Context 객체 생성 (기본값 null)
const AuthContext = createContext(null);

// 애플리케이션 전역에 인증 기능을 제공하는 Provider 컴포넌트
export function AuthProvider({children}) {
    // 사용자의 프로필 정보를 저장하는 상태
    const [user, setUser] = useState(null);
    // 앱 시작 시 토큰 확인 절차의 로딩 상태 관리
    const [isLoading, setIsLoading] = useState(true);
    // 현재 로그인 성공 여부를 나타내는 불리언 상태
    const [isAuthenticated, setIsAuthenticated] = useState(false);

    // 컴포넌트 마운트 시 브라우저 로컬스토리지에서 기존 인증 정보를 동기화
    useEffect(() => {
        const initAuth = async () => {
            // 로컬스토리지 내 저장된 데이터 추출
            const token = localStorage.getItem('token');
            const savedUser = localStorage.getItem('user');

            // 저장된 토큰과 사용자 정보가 모두 존재할 경우 인증 복구
            if (token && savedUser) {
                try {
                    // 문자열 기반 JSON 데이터를 객체로 변환하여 상태 반영
                    const userData = JSON.parse(savedUser);
                    setUser(userData);
                    setIsAuthenticated(true);
                } catch (error) {
                    // 데이터 파싱 실패 시 보안을 위해 관련 스토리지 완전 삭제
                    console.error('사용자 정보 파싱오류', error);
                    localStorage.removeItem('token');
                    localStorage.removeItem('user');
                }
            }
            // 초기 인증 확인 완료 후 로딩 상태 해제
            setIsLoading(false);
        }

        // 인증 초기화 함수 실행
        initAuth();
    }, []);
```

## 주요 인증 액션 로직
- **로그인 (login)**
    - 사용자 계정 정보 기반 API 통신 수행
    - 응답 데이터 내 토큰과 사용자 객체 저장
- **로그아웃 (logout)**
    - 서버 세션 종료 API 호출 및 로컬 저장소 완전 초기화
- **관리자 판별 (isAdmin)**
    - 사용자 역할(role) 데이터 확인을 통한 권한 검증
```javascript
    // 로그인 처리 함수: credentials(아이디/비밀번호 객체)를 받아 API 연동 및 성공 시 데이터 저장
    const login = useCallback(async (credentials) => {
        try {
            const data = await authApi.login(credentials);

            // 응답 데이터 내 유효한 토큰이 존재하는지 확인
            if (data.token) {
                // 토큰 및 사용자 정보를 문자열화하여 스토리지에 저장
                localStorage.setItem("token", data.token);
                localStorage.setItem('user', JSON.stringify(data.user || data));
                // 전역 상태 업데이트 및 인증 활성화
                setUser(data.user || data);
                setIsAuthenticated(true);

                return { success: true, data };
            }

            return { success: false, message: '로그인에 실패했습니다.' };
        } catch (error) {
            console.error('로그인 오류', error);
            return {
                success: false,
                message: error.response?.data?.message || '로그인에 실패했습니다.'
            }
        }
    }, []);

    // 로그아웃 처리 함수: 서버 알림 후 로컬 데이터 및 상태 초기화
    const logout = useCallback(async () => {
        try { 
            // 서버 측 로그아웃 처리 수행
            await authApi.logout();
        } catch (error) {
            console.error('로그아웃 API 오류 : ', error);
        } finally {
            // 결과와 상관없이 로컬 인증 정보 제거 및 상태 초기화
            localStorage.removeItem('token');
            localStorage.removeItem('user');
            setUser(null);
            setIsAuthenticated(false); 
        }
    }, []);
    
    // ✨ 추가: Axios와 AuthContext 상태 동기화
    useEffect(() => {
        const interceptor = api.interceptors.response.use(
            (response) => response, // 성공 시 통과
            (error) => {
                if (error.response?.status === 401) {
                    // 서버가 401(인증실패)을 던지면 리액트도 즉시 로그아웃!
                    alert("세션이 만료되어 자동으로 로그아웃됩니다.");
                    logout(); 
                }
                return Promise.reject(error);
            }
        );

        // 컴포넌트가 사라질 때 인터셉터 제거 (메모리 관리)
        return () => api.interceptors.response.eject(interceptor);
    }, [logout]);

    // 사용자 정보 업데이트 함수: 내 정보 수정 등의 상황에서 호출
    const updateUser = useCallback((userData) => {
        setUser(userData);
        localStorage.setItem('user', JSON.stringify(userData));
    }, []);

    // 관리자 여부 확인 함수: ROLE_ADMIN 권한 보유 판별
    const isAdmin = useCallback(() => {
        return user?.role === 'ROLE_ADMIN';
    }, [user]);

    // 서버 측 업로드 파일 접근을 위한 기본 경로
    const filePath = '/spring/upload/';

    // Context 하위 컴포넌트들에게 공유할 데이터 묶음
    const value = {
        user,
        isLoading,
        isAuthenticated,
        login,
        logout,
        updateUser,
        isAdmin,
        filePath
    };

    return (
        <AuthContext.Provider value={value}>
            {children}
        </AuthContext.Provider>
    );
}

// 커스텀 훅: 컴포넌트에서 인증 컨텍스트를 간편하게 호출하기 위한 용도
export function useAuth() {
    const context = useContext(AuthContext);
    // Provider 외부 호출 방지를 위한 에러 처리
    if (!context) {
        throw new Error('useContext 오류 발생! AuthProvider를 이용해 사용하세요!');
    }
    return context;
}

export default AuthContext;
```
