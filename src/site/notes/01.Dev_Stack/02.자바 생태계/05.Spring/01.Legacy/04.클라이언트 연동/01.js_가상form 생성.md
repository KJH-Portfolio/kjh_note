#### js를 이용해 view에서 Post요청하기
- js를 통해서 버튼을 누르면 form과 hidden input요소를 생성하고 요청을 보내는 방식
```java
	// jQuery의 $(function(){...})와 동일한 역할을 합니다.
	document.addEventListener('DOMContentLoaded', function() {
		
		// 1. 'updateBtn' ID를 가진 버튼 요소를 찾습니다.
		const updateBtn = document.getElementById('updateBtn');

		// 2. 버튼에 클릭 이벤트 리스너를 추가합니다.
		updateBtn.addEventListener('click', function() {
			
			// **3. POST 요청을 위한 가상의 <form> 태그를 생성합니다.**
			const form = document.createElement('form');
			form.action = 'update.bo';
			form.method = 'post';
			
			// **4. 서버로 전달할 데이터를 담을 <input type="hidden"> 태그를 생성합니다.**
			const input = document.createElement('input');
			input.type = 'hidden';
			input.name = 'bno';
			// 💡 주의: "${detail.boardNo}"는 JSP/템플릿 엔진 문법이므로 그대로 사용합니다.
			input.value = '${detail.boardNo}'; 
			
			// 5. <input> 태그를 <form> 태그 안에 삽입합니다.
			form.appendChild(input);
			
			// 6. <form> 태그를 문서의 <body> 영역에 일시적으로 추가합니다.
			document.body.appendChild(form);
			
			// **7. 가상의 <form>을 강제로 제출(submit)합니다.**
			form.submit();
			
			// 참고: 요청 전송 후, 생성한 폼 요소를 깔끔하게 제거할 수도 있습니다.
			// document.body.removeChild(form); 
		});
	});
```
